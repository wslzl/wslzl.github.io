JavaScript 常用

类型

判断类型

const isType = (type) => (target) =>
  `[object ${type}]` === Object.prototype.toString.call(target);

string 强制类型转换

"123" * 1 + "123";

  // Bool

  ~~num

num & 1; // 判断奇偶

数组

置空

array.length = 0;

填充

let initializeArray = (n, value = 0) => Array(n).fill(value);

克隆

const _arr = [];
const arr = [..._arr];

交换

let x = 3;
let y = 4;
[x, y] = [y, x];

合并

const arr1 = [];
const arr2 = [];

去重

const unique = (array) => Array.from(new Set(array));
const unique = (array) => [...new Set(array)];

计数

const count = (array, value) =>
  array.reduce((a, v) => (v === value ? a + 1 : a + 0), 0);

最值

const arrayMax = (array) => Math.max(...array);
const arrayMin = (array) => Math.min(...array);

相似

const similarity = (array, values) => array.filter((v) => values.includes(v));

求和

const total = (array) =>
  array.reduce((prev, cur) => {
    return prev + cur;
  });

平均

const average = (array) =>
  array.reduce((prev, cur) => prev + cur, 0) / array.length;

排序

const cnSort = (array) =>
  array.sort((a, b) => a.localeCompare(b, "zh-Hans-CN"), {
    sensitivity: "accent",
  });
